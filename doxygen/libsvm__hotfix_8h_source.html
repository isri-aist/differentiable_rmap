41 inline int svm_save_model_hotfix(const char * model_file_name, const svm_model * model)
42 {
43  FILE * fp = fopen(model_file_name, "w");
44  if(fp == NULL) return -1;
45 
46  char * old_locale = setlocale(LC_ALL, NULL);
47  if(old_locale)
48  {
49  old_locale = strdup(old_locale);
50  }
51  setlocale(LC_ALL, "C");
52 
53  const svm_parameter & param = model->param;
54 
55  fprintf(fp, "svm_type %s\n", svm_type_table[param.svm_type]);
56  fprintf(fp, "kernel_type %s\n", kernel_type_table[param.kernel_type]);
57 
58  if(param.kernel_type == POLY) fprintf(fp, "degree %d\n", param.degree);
59 
60  if(param.kernel_type == POLY || param.kernel_type == RBF || param.kernel_type == SIGMOID)
61  fprintf(fp, "gamma %.17g\n", param.gamma);
62 
63  if(param.kernel_type == POLY || param.kernel_type == SIGMOID) fprintf(fp, "coef0 %.17g\n", param.coef0);
64 
65  int nr_class = model->nr_class;
66  int l = model->l;
67  fprintf(fp, "nr_class %d\n", nr_class);
68  fprintf(fp, "total_sv %d\n", l);
69 
70  {
71  fprintf(fp, "rho");
72  for(int i = 0; i < nr_class * (nr_class - 1) / 2; i++) fprintf(fp, " %.17g", model->rho[i]);
73  fprintf(fp, "\n");
74  }
75 
76  if(model->label)
77  {
78  fprintf(fp, "label");
79  for(int i = 0; i < nr_class; i++) fprintf(fp, " %d", model->label[i]);
80  fprintf(fp, "\n");
81  }
82 
83  if(model->probA) // regression has probA only
84  {
85  fprintf(fp, "probA");
86  for(int i = 0; i < nr_class * (nr_class - 1) / 2; i++) fprintf(fp, " %.17g", model->probA[i]);
87  fprintf(fp, "\n");
88  }
89  if(model->probB)
90  {
91  fprintf(fp, "probB");
92  for(int i = 0; i < nr_class * (nr_class - 1) / 2; i++) fprintf(fp, " %.17g", model->probB[i]);
93  fprintf(fp, "\n");
94  }
95 
96  // hotfix: Since an invalid value is set in model->nSV, set the value manually only for two-class classification.
97  if(param.svm_type == C_SVC || param.svm_type == NU_SVC)
98  {
99  if(nr_class != 2) throw std::runtime_error("multiple classes are not supported in svm_save_model_hotfix");
100  fprintf(fp, "nr_sv");
101  for(int i = 0; i < nr_class; i++)
102  if(i == 0)
103  fprintf(fp, " %d", l);
104  else
105  fprintf(fp, " 0");
106  fprintf(fp, "\n");
107  }
108 
109  fprintf(fp, "SV\n");
110  const double * const * sv_coef = model->sv_coef;
111  const svm_node * const * SV = model->SV;
112 
113  for(int i = 0; i < l; i++)
114  {
115  for(int j = 0; j < nr_class - 1; j++) fprintf(fp, "%.17g ", sv_coef[j][i]);
116 
117  const svm_node * p = SV[i];
118 
119  if(param.kernel_type == PRECOMPUTED)
120  fprintf(fp, "0:%d ", (int)(p->value));
121  else
122  while(p->index != -1)
123  {
124  fprintf(fp, "%d:%.8g ", p->index, p->value);
125  p++;
126  }
127  fprintf(fp, "\n");
128  }
129 
130  setlocale(LC_ALL, old_locale);
131  free(old_locale);
132 
133  if(ferror(fp) != 0 || fclose(fp) != 0)
134  return -1;
135  else
136  return 0;
137 }
