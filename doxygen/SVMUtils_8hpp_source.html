295 template<>
296 inline SampleToSampleMat<SamplingSpace::SE2> relSampleToSampleMat<SamplingSpace::SE2>(
297  const Sample<SamplingSpace::SE2> & pre_sample,
298  const Sample<SamplingSpace::SE2> & suc_sample,
299  bool wrt_suc)
300 {
301  double cos = std::cos(pre_sample.z());
302  double sin = std::sin(pre_sample.z());
303 
304  SampleToSampleMat<SamplingSpace::SE2> mat;
305  mat << cos, sin, 0, -sin, cos, 0, 0, 0, 1;
306 
307  if(!wrt_suc)
308  {
309  Vel<SamplingSpace::SE2> sample_error = sampleError<SamplingSpace::SE2>(pre_sample, suc_sample);
310  mat *= -1;
311  mat(0, 2) = -sin * sample_error.x() + cos * sample_error.y();
312  mat(1, 2) = -cos * sample_error.x() - sin * sample_error.y();
313  }
314 
315  return mat;
316 }
