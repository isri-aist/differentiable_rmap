58 template<size_t N>
59 bool kNearestNeighbor(const Eigen::Matrix<double, N, 1> & test_sample,
60  size_t K,
61  const std::vector<Eigen::Matrix<double, N, 1>> & train_sample_list,
62  const std::vector<bool> & class_list)
63 {
64  // Set idx_dist_list which is top-K nearest samples and distances
65  std::vector<std::pair<int, double>> idx_dist_list(
66  K, std::make_pair<int, double>(-1, std::numeric_limits<double>::max()));
67  for(size_t i = 0; i < train_sample_list.size(); i++)
68  {
69  double dist = (train_sample_list[i] - test_sample).squaredNorm();
70  for(size_t j = 0; j < K; j++)
71  {
72  if(dist < idx_dist_list[j].second)
73  {
74  for(size_t k = K - 1; k > j; k--)
75  {
76  idx_dist_list[k] = idx_dist_list[k - 1];
77  }
78  idx_dist_list[j].first = static_cast<int>(i);
79  idx_dist_list[j].second = dist;
80  break;
81  }
82  }
83  }
84 
85  // Determine class by majority vote of top-K samples
86  int positive_num = 0;
87  for(const auto & idx_dist : idx_dist_list)
88  {
89  if(idx_dist.first < 0)
90  {
91  mc_rtc::log::error_and_throw<std::runtime_error>(
92  "[kNearestNeighbor] idx_dist_list contains uninitialized elements.");
93  }
94  if(class_list[idx_dist.first])
95  {
96  positive_num++;
97  }
98  else
99  {
100  positive_num--;
101  }
102  }
103  if(positive_num == 0)
104  {
105  mc_rtc::log::error_and_throw<std::runtime_error>(
106  "[kNearestNeighbor] Numbers of nearest points of positive and negative are the same.");
107  }
108  return positive_num > 0;
109 }
