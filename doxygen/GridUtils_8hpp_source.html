31 template<class DivideIdxsType, class DivideNumsType>
32 bool updateGridDivideIdxs(DivideIdxsType & divide_idxs,
33  const DivideNumsType & divide_nums,
34  const std::vector<int> & update_dims)
35 {
36  size_t i_end = update_dims.empty() ? divide_idxs.size() : update_dims.size();
37  for(size_t i = 0; i < i_end; i++)
38  {
39  size_t k = update_dims.empty() ? i : update_dims[i];
40  divide_idxs[k]++;
41  if(divide_idxs[k] == divide_nums[k] + 1)
42  {
43  // If there is a carry, the current digit value is set to zero
44  divide_idxs[k] = 0;
45  // If there is a carry at the top, exit the outer loop
46  if(i == i_end - 1)
47  {
48  return true;
49  }
50  }
51  else
52  {
53  // If there is no carry, it will end
54  break;
55  }
56  }
57  return false;
58 }

228 template<SamplingSpace SamplingSpaceType, class DivideNumsType>
229 void loopGrid(const DivideNumsType & divide_nums,
230  const GridPos<SamplingSpaceType> & grid_pos_min,
231  const GridPos<SamplingSpaceType> & grid_pos_range,
232  const GridFuncType<SamplingSpaceType> & func,
233  const std::vector<int> & update_dims,
234  const GridIdxs<SamplingSpaceType> & default_divide_idxs)
235 {
236  // Initialize divide_idxs with default value
237  GridIdxs<SamplingSpaceType> divide_idxs = GridIdxs<SamplingSpaceType>::Zero();
238  if(!update_dims.empty())
239  {
240  for(size_t i = 0; i < divide_idxs.size(); i++)
241  {
242  if(std::find(update_dims.begin(), update_dims.end(), i) == update_dims.end())
243  {
244  if(default_divide_idxs[i] < 0 || default_divide_idxs[i] > divide_nums[i])
245  {
246  mc_rtc::log::error_and_throw<std::runtime_error>(
247  "[loopGrid] default_divide_idxs[{}] is invalid. It should be 0 <= {} <= {}", i, default_divide_idxs[i],
248  divide_nums[i]);
249  }
250  divide_idxs[i] = default_divide_idxs[i];
251  }
252  }
253  }
254 
255  // Loop
256  GridPos<SamplingSpaceType> divide_ratios;
257  bool break_flag = false;
258  do
259  {
260  gridDivideIdxsToRatios(divide_ratios, divide_idxs, divide_nums);
261 
262  // ROS_INFO_STREAM("[loopGrid] idx: " << calcGridIdx(divide_idxs, divide_nums) <<
263  // ", divide_idxs: [" << vecToStr(divide_idxs) << "]");
264  func(calcGridIdx(divide_idxs, divide_nums), divide_ratios.cwiseProduct(grid_pos_range) + grid_pos_min);
265 
266  break_flag = updateGridDivideIdxs(divide_idxs, divide_nums, update_dims);
267 
268  } while(!break_flag);
269 }
